        if x.dim() == 3:
            x = x + self.drop_path(self.attention(self.ln_1(x)))
            x = x + self.drop_path(self.mlp(self.ln_2(x)))
            x = x.unsqueeze(1)
        else:
            z = x[:, -1, :, :]
            z = z + self.drop_path(self.attention(self.ln_1(z)))
            z = z + self.drop_path(self.mlp(self.ln_2(z)))
            x = torch.concat([x, z.unsqueeze(1)], dim=1)
        return x
